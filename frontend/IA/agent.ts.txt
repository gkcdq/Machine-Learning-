import React from "react";
import { GridHandle } from "../map/grid";
import { CellType } from "../map/cell";
import { parseMap } from "./utils";

// ─── Types ────────────────────────────────────────────────────────────────────

export interface MapData {
  start: [number, number] | null;
  exit: [number, number] | null;
  collectibles: [number, number][];
  walls: [number, number][];
  rows: number;
  cols: number;
  iaposeX: number;
  iaposeY: number;
  qTable?: Record<string, number[]>;
}

// ─── Configuration Q-Learning ─────────────────────────────────────────────────

const ALPHA         = 0.15;
const GAMMA         = 0.95;
const EPSILON_START = 0.4;
const EPSILON_MIN   = 0.05;
const HISTORY_SIZE  = 20;

// ─── État Global ──────────────────────────────────────────────────────────────

export let qTable: Record<string, number[]> = {};
let currentX = 0;
let currentY = 0;
let moveInterval: ReturnType<typeof setInterval> | null = null;
let count = 0;
let posHistory: string[]              = [];
let visitFreq: Record<string, number> = {};

// ─── Q-Table helpers ──────────────────────────────────────────────────────────

function getQKey(r: number, c: number, pickedCount: number): string {
  return `${r},${c}|${pickedCount}`;
}

function getQValues(r: number, c: number, pickedCount: number): number[] {
  const key = getQKey(r, c, pickedCount);
  if (!qTable[key]) qTable[key] = [0, 0, 0, 0];
  return qTable[key];
}

// ─── BFS avec cache ───────────────────────────────────────────────────────────

/**
 * Cache global : distCache[`${fromR},${fromC}`][`${toR},${toC}`] = distance
 * Précalculé une fois au démarrage, réutilisé à chaque step.
 */
let distCache: Record<string, Record<string, number>> = {};

/**
 * BFS depuis un point source — retourne toutes les distances depuis ce point.
 * Appelé UNE SEULE FOIS par point d'intérêt au démarrage.
 */
function bfsFromSource(
  map: CellType[][],
  srcR: number,
  srcC: number
): Record<string, number> {
  const rows = map.length;
  const cols = map[0]?.length ?? 0;
  const dist: Record<string, number> = {};
  const queue: [number, number, number][] = [[srcR, srcC, 0]];
  dist[`${srcR},${srcC}`] = 0;

  while (queue.length > 0) {
    const [r, c, d] = queue.shift()!;
    for (const [dr, dc] of [[-1,0],[1,0],[0,-1],[0,1]]) {
      const nr = r + dr, nc = c + dc;
      const key = `${nr},${nc}`;
      if (nr < 0 || nr >= rows || nc < 0 || nc >= cols) continue;
      if (key in dist) continue;
      if (map[nr][nc] === "wall") continue;
      dist[key] = d + 1;
      queue.push([nr, nc, d + 1]);
    }
  }
  return dist;
}

/**
 * Récupère la distance entre deux points depuis le cache.
 * Si le BFS de ce point source n'a pas encore été calculé, le calcule et le met en cache.
 */
function getCachedDist(
  map: CellType[][],
  fromR: number, fromC: number,
  toR: number,   toC: number
): number {
  const srcKey = `${fromR},${fromC}`;
  if (!distCache[srcKey]) {
    distCache[srcKey] = bfsFromSource(map, fromR, fromC);
  }
  return distCache[srcKey][`${toR},${toC}`] ?? Infinity;
}

// ─── Ordre optimal des collectibles (nearest-neighbor TSP) ───────────────────

/**
 * Calcule l'ordre de visite optimal des collectibles depuis une position de départ.
 * Utilise nearest-neighbor : à chaque étape, va au collectible non visité le plus proche.
 * Complexité O(N²) — rapide jusqu'à ~20 collectibles.
 *
 * Retourne la liste ordonnée des clés "r,c" à visiter.
 */
function computeOptimalOrder(
  map: CellType[][],
  fromR: number,
  fromC: number,
  collectibleKeys: string[],
  exitR: number,
  exitC: number
): string[] {
  const remaining = [...collectibleKeys];
  const ordered: string[] = [];
  let curR = fromR, curC = fromC;

  while (remaining.length > 0) {
    let bestIdx  = 0;
    let bestDist = Infinity;

    remaining.forEach((key, idx) => {
      const [cr, cc] = key.split(",").map(Number);
      const d = getCachedDist(map, curR, curC, cr, cc);
      if (d < bestDist) { bestDist = d; bestIdx = idx; }
    });

    const chosen = remaining.splice(bestIdx, 1)[0];
    ordered.push(chosen);
    [curR, curC] = chosen.split(",").map(Number);
  }

  return ordered;
}

// ─── Backend & Persistance ────────────────────────────────────────────────────

export async function saveSessionToDB(data: MapData): Promise<number | null> {
  try {
    const res = await fetch("/api/ia/session/save/", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        rows: data.rows,
        cols: data.cols,
        collectibles: data.collectibles.map(([r, c]) => [c, r]),
        q_table: qTable,
      }),
    });
    const json = await res.json();
    if (json.status === "ok") return json.session_id;
    return null;
  } catch {
    return null;
  }
}

export async function loadLastSession(): Promise<{ sessionId: number; collectibles: [number, number][] } | null> {
  try {
    const res  = await fetch("/api/ia/session/last/");
    const json = await res.json();
    if (json.status !== "ok") return null;
    const collectibles: [number, number][] = json.collectibles.map(
      (c: { x: number; y: number }) => [c.y, c.x] as [number, number]
    );
    return { sessionId: json.session_id, collectibles };
  } catch {
    return null;
  }
}

// ─── Logique de Déplacement & Entraînement ────────────────────────────────────

export function startMovement(
  gridRef: React.RefObject<GridHandle>,
  maxRows: number,
  maxCols: number,
  startX: number,
  startY: number
) {
  if (moveInterval) clearInterval(moveInterval);

  const initialMap = gridRef.current!.getMap();
  const collectiblesList: string[] = [];
  let exitR = -1, exitC = -1;

  initialMap.forEach((row, r) =>
    row.forEach((cell, c) => {
      if (cell === "collectible") collectiblesList.push(`${r},${c}`);
      if (cell === "exit") { exitR = r; exitC = c; }
    })
  );

  const totalCollectibles = collectiblesList.length;

  // ── Précalcul du cache BFS depuis tous les points d'intérêt ─────────────────
  // On calcule le BFS depuis : départ + chaque collectible + sortie
  // Ça se fait UNE SEULE FOIS avant le premier épisode
  distCache = {}; // reset si nouvelle map
  console.log("[IA] Précalcul BFS...");

  const sources: [number, number][] = [
    [startX, startY],
    ...collectiblesList.map(k => k.split(",").map(Number) as [number, number]),
    [exitR, exitC],
  ];
  sources.forEach(([r, c]) => {
    const key = `${r},${c}`;
    if (!distCache[key]) distCache[key] = bfsFromSource(initialMap, r, c);
  });

  console.log(`[IA] BFS précalculé pour ${sources.length} points.`);

  // ── Calcul de l'ordre optimal depuis le départ ───────────────────────────────
  const optimalOrder = computeOptimalOrder(
    initialMap, startX, startY, [...collectiblesList], exitR, exitC
  );
  console.log("[IA] Ordre optimal :", optimalOrder);

  // Init état épisode
  currentX = startX;
  currentY = startY;
  let picked         = new Set<string>();
  let currentEpsilon = EPSILON_START;
  let exitSeen       = false; // devient true dès que l'agent touche la sortie
  posHistory = [];
  visitFreq  = {};

  moveInterval = setInterval(() => {
    if (!gridRef.current) return;
    const map = gridRef.current.getMap();
    const pickedCount = picked.size;

    // ── 1. Choix de l'action (Epsilon-Greedy) ─────────────────────────────────
    const qValues = getQValues(currentX, currentY, pickedCount);
    const action  = Math.random() < currentEpsilon
      ? Math.floor(Math.random() * 4)
      : qValues.indexOf(Math.max(...qValues));

    let nr = currentX, nc = currentY;
    if (action === 0) nr--;
    if (action === 1) nr++;
    if (action === 2) nc--;
    if (action === 3) nc++;

    // ── 2. Calcul de la récompense ─────────────────────────────────────────────
    let reward = -1;
    const targetCell = map[nr]?.[nc];
    const nextPicked = new Set(picked);

    if (!targetCell || targetCell === "wall") {
      reward = -15;
      nr = currentX;
      nc = currentY;

    } else if (targetCell === "collectible") {
      const cellKey = `${nr},${nc}`;
      if (!picked.has(cellKey)) {
        reward = 300;
        nextPicked.add(cellKey);
        gridRef.current!.consumeCell(nr, nc);
      }

    } else if (targetCell === "exit") {
      if (picked.size === totalCollectibles) {
        reward = 1000;
      } else {
        reward = -100;
      }
    }

    // ── 3. Reward shaping avec cache BFS + ordre optimal ──────────────────────
    if (nr !== currentX || nc !== currentY) {

      if (pickedCount < totalCollectibles) {
        // Objectif principal : prochain collectible dans l'ordre optimal
        const nextTarget = optimalOrder.find(k => !picked.has(k));
        if (nextTarget) {
          const [tr, tc] = nextTarget.split(",").map(Number);
          const db = distCache[`${currentX},${currentY}`]?.[`${tr},${tc}`] ?? Infinity;
          const da = distCache[`${nr},${nc}`]?.[`${tr},${tc}`] ?? Infinity;
          if (db !== Infinity && da !== Infinity) {
            reward += (db - da) * 4;
          }
        }

        // Objectif secondaire (poids faible) : garder la sortie en mémoire
        // L'agent accumule un biais directionnel vers la sortie même en ramassant
        if (exitR >= 0 && exitSeen) {
          const db = distCache[`${currentX},${currentY}`]?.[`${exitR},${exitC}`] ?? Infinity;
          const da = distCache[`${nr},${nc}`]?.[`${exitR},${exitC}`] ?? Infinity;
          if (db !== Infinity && da !== Infinity) {
            reward += (db - da) * 1; // poids faible pour ne pas court-circuiter les collectibles
          }
        }

      } else if (exitR >= 0) {
        // Tous ramassés → guider fortement vers la sortie (déjà vue ou non)
        const db = distCache[`${currentX},${currentY}`]?.[`${exitR},${exitC}`] ?? Infinity;
        const da = distCache[`${nr},${nc}`]?.[`${exitR},${exitC}`] ?? Infinity;
        if (db !== Infinity && da !== Infinity) {
          reward += (db - da) * 6; // poids plus fort car c'est l'objectif final
        }
      }
    }

    // ── Mémoire de la sortie : marque dès le premier contact ──────────────────
    if (targetCell === "exit" && !exitSeen) {
      exitSeen = true;
      console.log(`[IA] Sortie mémorisée en [${nr},${nc}]`);
    }

    // ── 4. Anti-cycle ─────────────────────────────────────────────────────────
    const posKey = `${nr},${nc}`;

    const recentCount = posHistory.filter(p => p === posKey).length;
    if (recentCount >= 2) reward -= 10 * recentCount;

    visitFreq[posKey] = (visitFreq[posKey] ?? 0) + 1;
    if (visitFreq[posKey] > 6) {
      qValues[action] -= 15 * (visitFreq[posKey] - 6);
    }

    posHistory.push(posKey);
    if (posHistory.length > HISTORY_SIZE) posHistory.shift();

    // ── 5. Apprentissage Bellman ───────────────────────────────────────────────
    const nextQValues = getQValues(nr, nc, nextPicked.size);
    const maxNextQ    = Math.max(...nextQValues);
    qValues[action]  += ALPHA * (reward + GAMMA * maxNextQ - qValues[action]);

    // ── 6. Mise à jour état physique ──────────────────────────────────────────
    currentX = nr;
    currentY = nc;
    picked   = nextPicked;
    gridRef.current.moveAgent(currentX, currentY);

    // ── 7. Décroissance epsilon ────────────────────────────────────────────────
    if (currentEpsilon > EPSILON_MIN) currentEpsilon *= 0.9998;

    // ── 8. Reset épisode ──────────────────────────────────────────────────────
    if (targetCell === "exit" && picked.size === totalCollectibles) {
      count++;
      console.log(`Épisode ${count} | Epsilon : ${currentEpsilon.toFixed(3)}`);

      currentX   = startX;
      currentY   = startY;
      picked     = new Set<string>();
      posHistory = [];
      visitFreq  = {};
      exitSeen   = true; // déjà vue au moins une fois — on garde la mémoire entre épisodes

      collectiblesList.forEach((key) => {
        const [r, c] = key.split(",").map(Number);
        gridRef.current!.restoreCell(r, c, "collectible");
      });

      gridRef.current.moveAgent(startX, startY);
    }
  }, 0);
}

export function stopRandomMovement() {
  if (moveInterval) {
    clearInterval(moveInterval);
    moveInterval = null;
  }
}

export async function runAgent(gridRef: React.RefObject<GridHandle>) {
  if (!gridRef.current) return;
  const map  = gridRef.current.getMap();
  const data = parseMap(map);

  if (!data.start) return;
  const [startRow, startCol] = data.start;

  await loadLastSession();
  startMovement(gridRef, data.rows, data.cols, startRow, startCol);
  saveSessionToDB(data);
}
